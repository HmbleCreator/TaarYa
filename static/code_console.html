import astropy.units as u
from astropy.coordinates import SkyCoord
from astroquery.gaia import Gaia
# Initialize TaarYa autonomous API query for a specific cone search
def cone_search(ra, dec, radius):
# Simulate cone search logic targeting the local backend
coord = SkyCoord(ra=ra, dec=dec, unit=(u.deg, u.deg), frame='icrs')
print(f"Searching at RA: {ra}, DEC: {dec} with radius {radius} deg")
query = f"""
SELECT source_id, ra, dec, parallax, phot_g_mean_mag
FROM gaiaedr3.gaia_source
WHERE CONTAINS(POINT('ICRS', ra, dec),
CIRCLE('ICRS', {coord.ra.deg}, {coord.dec.deg}, {radius})) = 1
"""
job = Gaia.launch_job_async(query)
return job.get_results()
# Search for Proxima Centauri
