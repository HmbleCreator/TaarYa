import numpy as np
import astropy.units as u
from astropy.coordinates import SkyCoord
from astroquery.gaia import Gaia
# Define target coordinates for globular cluster NGC 4590 (M68)
target_coord = SkyCoord(ra=189.866*u.deg, dec=-26.744*u.deg, frame='icrs')
def query_gaia_cone(coord, radius_deg=0.5):
    """
    Perform asynchronous cone search on Gaia DR3 source table.
    Filters for proper motion coherence typical of cluster members.
    """
    width = u.Quantity(radius_deg, u.deg)
    height = u.Quantity(radius_deg, u.deg)

    # Agent Note: Using async job to handle large dataset return
    job = Gaia.launch_job_async(f"""
    SELECT source_id, ra, dec, parallax, pmra, pmdec, phot_g_mean_mag
    FROM gaiadr3.gaia_source
    WHERE 1=CONTAINS(
        POINT('ICRS', ra, dec),
        CIRCLE('ICRS', {coord.ra.deg}, {coord.dec.deg}, {radius_deg})
    )
    AND parallax > 0.1 AND parallax < 5.0
    """)

    return job.get_results()
# Execute Query
results = query_gaia_cone(target_coord)
